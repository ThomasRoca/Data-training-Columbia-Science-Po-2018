	// Read first cell
	Table[0].cells[0].innerText
	// create an empty array
	var array=[];
	// loop
	for(i=0; i<Table.length; i++) {
	array[i]={name:"", data:[]};
	//Copy the first column
	array[i].name=Table[i].cells[1].innerText;
